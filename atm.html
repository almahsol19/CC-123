The initial balance is **PHP 100**, and the system ensures **secure transactions** with input validation.

#include <iostream>
using namespace std;

int main() {
    int pin = 1234, enteredPin, attempts = 0;
    double balance = 100.0; // Initial balance
    double deposit, withdraw;
    int choice;

    // Allow up to 3 attempts for PIN entry
    while (attempts < 3) {
        cout << "üî¢ Enter your ATM PIN: ";
        cin >> enteredPin;

        if (enteredPin == pin) {
            cout << "‚úÖ PIN Verified Successfully!\n";
            break;
        }

        attempts++;
        if (attempts == 1) {
            cout << "‚ùå Incorrect PIN. üîÑ 2nd attempt remaining.\n";
        } else if (attempts == 2) {
            cout << "‚ùå Incorrect PIN. üö® 3rd attempt remaining.\n";
        } else {
            cout << "üö´ Too many incorrect attempts! Exiting...\n";
            return 0;
        }
    }

    do {
        // Display menu
        cout << "\nüí≥ Cyra's ATM Machine Menu:\n";
        cout << "1Ô∏è‚É£ Check Balance\n";
        cout << "2Ô∏è‚É£ Deposit Money\n";
        cout << "3Ô∏è‚É£ Withdraw Money\n";
        cout << "4Ô∏è‚É£ Exit\n";
        cout << "üìå Enter your choice: ";
        cin >> choice;

        switch (choice) {
            case 1: // Check Balance
                cout << "üí∞ Current Balance: PHP " << balance << endl;
                break;

            case 2: // Deposit Money
                cout << "üíµ Enter amount to deposit: PHP ";
                cin >> deposit;
                balance += deposit;
                cout << "üì• Amount Deposited: PHP " << deposit << endl;
                break;

            case 3: // Withdraw Money
                cout << "üí∏ Enter amount to withdraw: PHP ";
                cin >> withdraw;
                if (withdraw > balance) {
                    cout << "‚ùó Insufficient balance! ‚ùå\n";
                } else {
                    balance -= withdraw;
                    cout << "üì§ Amount Withdrawn: PHP " << withdraw << endl;
                }
                break;

            case 4: // Exit
                cout << "üôè Thank you for using the ATM. Have a great day! üöÄ\n";
                break;

            default:
                cout << "‚ö†Ô∏è Invalid choice! Please try again.\n";
        }

        cout << endl; // New line for better readability
    } while (choice != 4);

    return 0;
}
